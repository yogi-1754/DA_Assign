         CREATE DEFINER=`root`@`localhost` PROCEDURE `GET_COUNTRY_PAYMENTS`(IN YEAR INT, IN COUNTRY VARCHAR(50), OUT TOTALAMOUNT VARCHAR(50))
         BEGIN
         DECLARE TOTAL INT;

         SELECT ROUND(SUM(PAYMENTS.AMOUNT) / 1000)  INTO TOTAL
         FROM PAYMENTS JOIN CUSTOMERS
         ON PAYMENTS.CUSTOMERNUMBER = CUSTOMERS.CUSTOMERNUMBER
         WHERE YEAR(PAYMENTS.PAYMENTDATE) = YEAR AND CUSTOMERS.COUNTRY = COUNTRY;

         SET TOTALAMOUNT = CONCAT(TOTAL,'K');


         END
